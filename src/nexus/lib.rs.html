1//! Bindings for Raidcore Nexus addons.
2//!
3//! # Usage
4//! ```no_run
5//! # mod main {
6//! use nexus::{
7//!     gui::{register_render, render, RenderType},
8//!     imgui::Window
9//! };
10//!
11//! nexus::export! {
12//!     name: "My Addon",
13//!     signature: -0x12345678,
14//!     load: || {
15//!         register_render(RenderType::Render, render!(|ui| {
16//!             Window::new("My Window").build(ui, || {
17//!                 ui.text("Hello World");
18//!             });
19//!         }));
20//!     }
21//! }
22//! # }
23//! ```

43/// Returns the Nexus [`AddonApi`] instance.
44///
45/// Panics if called before initialization.

52/// Fields supported by the [`export`] macro.
53pub struct SupportedFields {
54    /// Raidcore addon id or random unique negative integer, if not on Raidcore.
55    pub signature: i32,
56
57    /// Name of the addon. Defaults to `CARGO_PKG_NAME`.
58    pub name: Option<String>,
59
60    /// Load function of the addon.
61    pub load: Option<AddonLoad>,
62
63    /// Unload function of the addon.
64    pub unload: Option<AddonUnload>,
65
66    /// Information about the addon.
67    pub flags: Option<AddonFlags>,
68
69    /// What platform the addon is hosted on.
70    pub provider: Option<UpdateProvider>,
71
72    /// Link to the update resource.
73    pub update_link: Option<&'static str>,
74
75    #[cfg(feature = "log")]
76    /// Filter for the log. Same syntax as [env_logger](https://docs.rs/env_logger/latest/env_logger/#enabling-logging).
77    pub log_filter: Option<&'static str>,
78}
