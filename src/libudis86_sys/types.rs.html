#[repr(C)]
#[derive(Clone, Copy)]
pub union ud_lval {
    pub sbyte: i8,
    pub ubyte: u8,
    pub sword: i16,
    pub uword: u16,
    pub sdword: i32,
    pub udword: u32,
    pub sqword: i64,
    pub uqword: u64,
    pub ptr: ud_lval_ptr,
}

impl fmt::Debug for ud_lval {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        write!(f, "ud_lval {{ union }}")
    }
}

#[repr(C)]
#[derive(Debug, Copy, Clone)]
pub struct ud_lval_ptr {
    pub seg: u16,
    pub off: u32,
}
