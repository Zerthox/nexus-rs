Crate nexus

Source
Expand description

Bindings for Raidcore Nexus addons.

§Usage

use nexus::{
    gui::{register_render, render, RenderType},
    imgui::Window
};

nexus::export! {
    name: "My Addon",
    signature: -0x12345678,
    load: || {
        register_render(RenderType::Render, render!(|ui| {
            Window::new("My Window").build(ui, || {
                ui.text("Hello World");
            });
        }));
    }
}
