Module nexus::event

source ·
Expand description

Event system.

use nexus::{
    event::{ADDON_LOADED, event_consume},
    log::{log, LogLevel}
};

let callback = event_consume!(|payload: Option<&i32>| {
    if let Some(signature) = payload {
        log(LogLevel::Info, "My Addon", format!("Addon {signature:?} loaded"));
    }
});

ADDON_LOADED.subscribe(callback);
